USEARCH commands for ITS on linux

## you can change the error threshold (this is the 'predicted error' whatever that means) to 2% or 3% using the fasq_maxee, default is 1%.

worsi@wla156:~/Svalbard_ITS$ for fq in *.fastq; do    ./usearch -fastq_filter $fq -fastq_maxee 1.0 -relabel @ -fastaout $fq.labeled.fa; done

worsi@wla156:~/Svalbard_ITS$ for fq in *.fastq; do   ./usearch -fastq_filter $fq -fastq_maxee 1.0 -relabel Filt -fastaout $fq.filtered.fa; done

worsi@wla156:~/Svalbard_ITS$ cat *.labeled.fa > labeled.fa

worsi@wla156:~/Svalbard_ITS$ cat *.filtered.fa > filtered.fa

worsi@wla156:~/Svalbard_ITS$ ./usearch -fastx_uniques filtered.fa -relabel Uniq -sizeout -fastaout uniques.fa

##RUN THE DEFAULT, ALWAYS CLUSTERS at 97% not possible to change.  Chimera checking included

./usearch -cluster_otus uniques.fa -minsize 2 -otus otus.fa -relabel Otu

###CLUSTER AT A DIFFERENT LEVEL, USER DEFINED, NO CHIMERA CHECKING

worsi@wla156:~/Svalbard_ITS$ ./usearch -cluster_fast uniques.fa -minsize 4 -centroids otus.fa -relabel Otu -id 0.95

## make the OTU table, if using cluster_fast then update the -id to match the threshold

worsi@wla156:~/Svalbard_ITS$ ./usearch -usearch_global labeled.fa -db otus.fa -strand plus -id 0.97 -otutabout otutab.txt -biomout otutab.json


#### The commands up until this point make an OTU table with no taxonomy.  Run BLASTn of the OTU sequences (output from USEARCH) against the RefSeq+UNITE database that Bill has already prepared (UNITE_and_RefSeq_BLASTdb).

 blastn -db ../../../../BLAST_databases/UNITE_and_RefSeq_combined/UNITE_and_RefSeq_BLASTdb -query ../otus.fa -out Svalbard_ITS_all3runscombined_BLASTn_UNITE_and_RefSeq_RESULTS.txt -outfmt 6 -max_target_seqs 1
